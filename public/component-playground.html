Web Component Starter

One frontend component, multiple host environments.

This starter uses a standard custom element so the same component can be dropped into WordPress, React, Angular, or a plain HTML page without rewriting the view layer.

Controls

Shape the API

These fields map directly to the component attributes. The body copy and action buttons are passed in as slotted content from the host page.

Usage

Drop the same tag into each host

Vanilla HTML

<script src="/components/byda-components.js" defer></script>

<byda-status-card
  heading="Plain HTML host"
  status="Ready"
  tone="success"
>
  <button slot="actions">Open enquiry</button>
</byda-status-card>

React

import "../dist/components/byda-components.esm.js";

export function Card() {
  return (
    <byda-status-card heading="React host" status="Ready">
      <button slot="actions">Open enquiry</button>
    </byda-status-card>
  );
}

Angular

import "../dist/components/byda-components.esm.js";

@NgModule({
  schemas: [CUSTOM_ELEMENTS_SCHEMA],
})

WordPress

wp_enqueue_script(
  "byda-components",
  get_stylesheet_directory_uri() . "/assets/byda-components.js",
  array(),
  null,
  true
);
